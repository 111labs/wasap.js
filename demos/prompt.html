wasap.init({
  elementCallback: function(el, wasap_send){
    el.addEventListener('click', function(e){
      e.preventDefault();
      var prompt_message = prompt('Say hello!', 'Hello from wasap.js');
      if(!prompt_message) return;
      wasap_send({ text: prompt_message });
    });
  }
})
